#### 2.1 项目概况对象 → 文本

```javascript
// 对象转文本（键值对格式）
const projectInfoToText = (projectInfo) => {
  if (!projectInfo) return ''

  const fieldMap = {
    projectName: '项目名称',
    constructionUnit: '建设单位',
    designUnit: '设计单位',
    projectAddress: '项目地址',
    buildingArea: '建筑面积',
    buildingHeight: '建筑高度',
    buildingType: '建筑类型',
    structureType: '结构类型',
    floors: '地上层数',
    undergroundFloors: '地下层数',
    seismicIntensity: '抗震设防烈度',
    climateZone: '建筑气候分区',
    landUseType: '土地使用性质',
    indoorOutdoorHeight: '室内外高差',
    absoluteElevation: '绝对标高',
    buildingFunction: '使用功能',
    fireResistanceGrade: '耐火等级',
    fireCategory: '防火类别',
    waterproofGrade: '防水等级',
    energyEfficiencyRequirement: '节能要求',
    barrierFreeRequirement: '无障碍要求'
  }

  return Object.entries(fieldMap)
    .map(([key, label]) => {
      const value = projectInfo[key]
      return value ? `${label}: ${value}` : null
    })
    .filter(Boolean)
    .join('\n')
}

// 文本转对象
const textToProjectInfo = (text) => {
  const reverseMap = {
    '项目名称': 'projectName',
    '建设单位': 'constructionUnit',
    '设计单位': 'designUnit',
    '项目地址': 'projectAddress',
    '建筑面积': 'buildingArea',
    '建筑高度': 'buildingHeight',
    '建筑类型': 'buildingType',
    '结构类型': 'structureType',
    '地上层数': 'floors',
    '地下层数': 'undergroundFloors',
    '抗震设防烈度': 'seismicIntensity',
    '建筑气候分区': 'climateZone',
    '土地使用性质': 'landUseType',
    '室内外高差': 'indoorOutdoorHeight',
    '绝对标高': 'absoluteElevation',
    '使用功能': 'buildingFunction',
    '耐火等级': 'fireResistanceGrade',
    '防火类别': 'fireCategory',
    '防水等级': 'waterproofGrade',
    '节能要求': 'energyEfficiencyRequirement',
    '无障碍要求': 'barrierFreeRequirement'
  }

  const result = {}
  text.split('\n').forEach(line => {
    const colonIndex = line.indexOf(':')
    if (colonIndex > 0) {
      const label = line.substring(0, colonIndex).trim()
      const value = line.substring(colonIndex + 1).trim()
      const key = reverseMap[label]
      if (key && value) {
        result[key] = value
      }
    }
  })

  return result
}
```

### 第三步：重构生成逻辑（45分钟）

**位置**：替换现有的 `sendMessage` 函数

```javascript
// 表单验证
const canGenerate = computed(() => {
  return (
    props.currentChapter?.id &&         // 必须选择章节
    projectInfoText.value.trim() &&     // 必须有项目概况
    chapterTemplateText.value.trim() && // 必须有章节模板
    !isGenerating.value                 // 不在生成中
  )
})

// 构造生成提示词
const buildGenerationPrompt = () => {
  const parts = []

  // 基础任务描述
  parts.push(`请根据以下信息生成《${props.currentChapter?.title || ''}》章节的施工图设计说明。`)

  // 补充要求
  if (additionalRequirements.value.trim()) {
    parts.push(`\n【特殊要求】\n${additionalRequirements.value}`)
  }

  // 输出格式要求
  parts.push('\n【输出格式】\n请使用 Markdown 格式输出，符合国家建筑规范标准。')

  return parts.join('\n')
}

// 生成章节内容（替换 sendMessage）
const handleGenerate = async () => {
  if (!canGenerate.value) return

  // 1. 折叠表单，准备显示生成结果
  formPanelExpanded.value = false
  messages.value = []
  stepIdToIndex.value.clear()

  // 2. 创建 assistant 消息占位符
  const assistantMsgId = Date.now().toString()
  const assistantMsg = {
    id: assistantMsgId,
    role: 'assistant',
    content: '',
    steps: []
  }
  messages.value.push(assistantMsg)
  const reactiveAssistantMsg = messages.value[0]

  isGenerating.value = true
  scrollToBottom()

  // 3. 创建 AbortController
  if (abortController.value) {
    abortController.value.abort()
  }
  abortController.value = new AbortController()

  try {
    let fullContent = ''

    // 4. 调用 workflow API（新增 template 和 project_info 参数）
    await workflowChatStream(
      {
        message: buildGenerationPrompt(),
        project_id: props.projectId,
        template: chapterTemplateText.value,      // 新增
        project_info: projectInfoText.value,      // 新增
        user_id: 'current_user'
      },
      (event) => {
        // 复用现有的事件处理逻辑
        handleEvent(reactiveAssistantMsg, event.event, event.data)

        // 累积完整内容
        if (event.event === 'token' && event.data.content) {
          fullContent += event.data.content
        }
      },
      abortController.value.signal
    )

    // 5. 生成完成后自动应用到编辑器
    if (fullContent) {
      emit('content-generated', fullContent)
      ElMessage.success('章节内容已生成并应用到编辑器')
    }

  } catch (error) {
    console.error('生成失败:', error)
    const isAbort = error?.name === 'AbortError' || String(error).includes('AbortError')
    if (!isAbort) {
      reactiveAssistantMsg.content += `\n\n**错误:** ${error.message}`
      ElMessage.error('章节生成失败')
    }
  } finally {
    isGenerating.value = false
    // Ensure no step stays stuck in progress
    reactiveAssistantMsg.steps.forEach(s => {
      if (s.status === 'in_progress') {
        s.status = 'completed'
      }
    })
    scrollToBottom()
  }
}
```
